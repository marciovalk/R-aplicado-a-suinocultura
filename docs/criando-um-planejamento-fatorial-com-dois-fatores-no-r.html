Capítulo 7 Criando um planejamento fatorial com dois fatores no R

7.1 Exemplo da bateria

D <- expand.grid(MT = 1:3, T = c(15, 70, 125))
D
##   MT   T
## 1  1  15
## 2  2  15
## 3  3  15
## 4  1  70
## 5  2  70
## 6  3  70
## 7  1 125
## 8  2 125
## 9  3 125
D <- rbind(D, D, D, D)
set.seed(2591)
D <- D[order(sample(1:36)), ]
BatteryDes <- D[ c( "MT", "T" )]
BatteryDes
write.csv(BatteryDes, file = "BatteryDes.csv", row.names = FALSE)
